③ SearchHandler.java
package com.example;

import com.sap.cds.Result;
import com.sap.cds.ql.Select;
import com.sap.cds.services.handler.EventHandler;
import com.sap.cds.services.handler.annotations.On;
import com.sap.cds.services.handler.annotations.ServiceName;
import org.springframework.stereotype.Component;

@Component
@ServiceName("SearchService")
public class SearchHandler implements EventHandler {

    @On(event = "searchEmployee")
    public Result searchEmployee(SearchServiceContext context) {

        String keyword = (String) context.getParameterInfo()
                .getQueryParameter("keyword");

        return context.getService().run(
            Select.from("my.company.Employees")
                  .where(e -> e.get("Name").like("%" + keyword + "%"))
        );
    }
}

⑥ Main.controller.js
sap.ui.define([
    "sap/ui/core/mvc/Controller"
], function (Controller) {
    "use strict";

    return Controller.extend("searchapp.controller.Main", {

        onSearch: async function () {

            const oModel = this.getView().getModel();
            const sKeyword = this.byId("inputKeyword").getValue();

            const oBinding = oModel.bindContext("/searchEmployee(...)");

            oBinding.setParameter("keyword", sKeyword);

            await oBinding.execute();

            const aResult = oBinding.getBoundContext().getObject();

            console.log(aResult);
        }

    });
});

SearchHandler.java
@Component
@ServiceName("SearchService")
public class SearchHandler implements EventHandler {

    @On(event = "createEmployee")
    public void createEmployee(CreateEmployeeContext context) {

        Map<String, Object> data = context.getData();

        String name  = (String) data.get("Name");
        String email = (String) data.get("Email");
        String dept  = (String) data.get("Dept");

        context.getService().run(
            Insert.into("my.company.Employees")
                  .entry(Map.of(
                      "Name",  name,
                      "Email", email,
                      "Dept",  dept
                  ))
        );

        context.setResult("登録しました");
    }
}

2. Java backend
@Component
@ServiceName("SearchService")
public class SearchHandler implements EventHandler {

    @On(event = "createEmployee")
    public void createEmployee(CreateEmployeeContext context) {

        EmployeeInput input = context.getData();

        context.getService().run(
            Insert.into("my.company.Employees")
                  .entry(Map.of(
                      "Name", input.getName(),
                      "Email", input.getEmail(),
                      "Dept", input.getDept()
                  ))
        );

        context.setResult("登録しました");
    }
}

4. controller.js 完成例
sap.ui.define([
    "sap/ui/core/mvc/Controller",
    "sap/ui/model/json/JSONModel"
], function (Controller, JSONModel) {
    "use strict";

    return Controller.extend("searchapp.controller.Main", {

        onInit: function () {
            this.getView().setModel(
                new JSONModel({
                    Name: "",
                    Email: "",
                    Dept: ""
                }),
                "localModel"
            );
        },

        onCreate: async function () {

            const oData = this.getView().getModel("localModel").getData();

            const oModel = this.getView().getModel();

            const oBinding = oModel.bindContext("/createEmployee(...)");

            oBinding.setParameter("data", oData);

            await oBinding.execute();

            const sResult = oBinding.getBoundContext().getObject();

            console.log(sResult);
        }

    });
});
