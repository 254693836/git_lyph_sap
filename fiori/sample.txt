onInit: function() {
    var oCrossAppNav = this.getOwnerComponent().getCrossAppNavigation();
    var sMode = oCrossAppNav.getCurrentIntent().getParameter("Mode");
    
    var oViewModel = new sap.ui.model.json.JSONModel({
        mode: sMode
    });
    this.getView().setModel(oViewModel, "ui");
}

        enabled="{= ${ui>/mode} !== 'ReadOnly' }"
